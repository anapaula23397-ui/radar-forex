async function atualizar(){ let resposta = await fetch("/sinais") let dados = await resposta.json() let tabela = document.getElementById("tabela") tabela.innerHTML = "PARSINALHORA" dados.forEach(d => { let classe = "aguardar" if(d.sinal=="COMPRA") classe="compra" if(d.sinal=="VENDA") classe="venda" tabela.innerHTML += ` ${d.par} ${d.sinal} ${d.hora} ` }) } /* SINCRONIZA COM A PRÓXIMA VELA M1 */ function iniciarRadar(){ atualizar() let agora = new Date() let segundos = agora.getSeconds() let espera = (60 - segundos) * 1000 setTimeout(() => { atualizar() setInterval(atualizar,60000) }, espera) } iniciarRadar()
