async function atualizar() { let resposta = await fetch("/sinais") let dados = await resposta.json() let tabela = document.getElementById("tabela") tabela.innerHTML = "PARSINALHORA" dados.forEach(d => { let classe = "aguardar" if (d.sinal == "COMPRA") classe = "compra" if (d.sinal == "VENDA") classe = "venda" tabela.innerHTML += ` ${d.par} ${d.sinal} ${d.hora} ` }) } /* ATUALIZA A CADA 60 SEGUNDOS */ setInterval(atualizar, 60000) atualizar()
